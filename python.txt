import torch
import numpy as np
from modules.xfeat import XFeat

# 1. Initialize Model
model = XFeat().eval().cuda()

# 2. Create Dummy Input (Standard 640x480 for Jetson)
dummy_input = torch.randn(1, 1, 480, 640).cuda()

# 3. Export to ONNX
print("Exporting xFeat to ONNX (FP32)...")
torch.onnx.export(model, dummy_input, "xfeat.onnx", 
                  input_names=['input'], output_names=['output'],
                  opset_version=12)

# 4. Optional: FP16 Conversion (Using ONNX Runtime tools)
# If you have onnxconverter-common installed:
# pip3 install onnxconverter-common
try:
    from onnxconverter_common import float16
    import onnx
    print("Converting to FP16 for Jetson Nano speed boost...")
    model_onnx = onnx.load("xfeat.onnx")
    model_fp16 = float16.convert_float_to_float16(model_onnx)
    onnx.save(model_fp16, "xfeat_fp16.onnx")
    print("SUCCESS: xfeat_fp16.onnx created!")
except ImportError:
    print("Note: Install 'onnxconverter-common' to enable half-precision.")
import cv2
import numpy as np
import onnxruntime as ort
import time

# 1. Load ONNX with TensorRT Provider
# This tells the Jetson to build a hardware-specific engine
providers = [
    ('TensorRTExecutionProvider', {
        'device_id': 0,
        'trt_fp16_enable': True,  # This is the Quantization trigger!
        'trt_engine_cache_enable': True,
        'trt_engine_cache_path': './'
    }),
    'CUDAExecutionProvider'
]

print("Initializing TensorRT Engine... (This may take a minute on first run)")
session = ort.InferenceSession("xfeat_fp16.onnx", providers=providers)

# 2. Camera Setup
cap = cv2.VideoCapture(0)
cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

while True:
    ret, frame = cap.read()
    if not ret: break
    
    start = time.time()
    
    # Pre-processing: Grayscale and Normalize
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    inp = gray.astype(np.float32) / 255.0
    inp = inp[None, None, :, :] # Add batch and channel dims

    # Inference (The Quantized Speed)
    outputs = session.run(None, {'input': inp})
    
    # ... (Feature extraction logic here) ...

    fps = 1.0 / (time.time() - start)
    cv2.putText(frame, f"QUANTIZED FPS: {fps:.1f}", (20, 50), 1, 2, (0, 255, 0), 2)
    cv2.imshow("Jetson Nano Optimized SLAM", frame)
    
    if cv2.waitKey(1) == ord('q'): break

cap.release()

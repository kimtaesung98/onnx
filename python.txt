import onnxruntime as ort

# 사용할 프로바이더 순서 정의 (TensorRT를 1순위로)
providers = [
    ('TensorRTExecutionProvider', {
        'device_id': 0,
        'trt_fp16_enable': True,  # 오린 나노 성능 최적화
        'trt_engine_cache_enable': True,
        'trt_engine_cache_path': './model_cache'
    }),
    'CUDAExecutionProvider'
]

try:
    session = ort.InferenceSession("your_model.onnx", providers=providers)
    print("✅ TensorRT 활성화 성공!")
except Exception as e:
    print(f"❌ 설정 실패: {e}")
    # 현재 사용 가능한 프로바이더 목록 출력
    print(f"사용 가능 목록: {ort.get_available_providers()}")

export LD_LIBRARY_PATH=/usr/lib/aarch64-linux-gnu:$LD_LIBRARY_PATH

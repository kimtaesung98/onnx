# CUDA 및 cuDNN 경로 추가
export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/usr/local/cuda/lib64:/usr/lib/aarch64-linux-gnu

# 가상환경 사용 중이라면 바로 적용되도록 설정
source ~/.bashrc

import cv2
import numpy as np
import onnxruntime as ort
import time

# 1. TensorRT 설정
providers = [
    ('TensorrtExecutionProvider', {
        'device_id': 0,
        'trt_fp16_enable': True,
        'trt_engine_cache_enable': True,
        'trt_engine_cache_path': './'
    }),
    ('CUDAExecutionProvider', {'device_id': 0})
]

print("TensorRT 엔진 초기화 중...")
try:
    session = ort.InferenceSession("xfeat_fp16.onnx", providers=providers)
    print(f"활성화된 엔진: {session.get_providers()[0]}")
except Exception as e:
    print(f"오류 발생: {e}")
    exit()

cap = cv2.VideoCapture(0)

def preprocess(img):
    img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    # 모델이 float16이므로 입력도 반드시 float16이어야 함
    img = img.astype(np.float16) / 255.0 
    return img[None, None, :, :]

while True:
    ret, frame = cap.read()
    if not ret: break
    
    start = time.time()
    
    # 입력 텐서 생성
    input_tensor = preprocess(frame)
    
    # Inference
    outputs = session.run(None, {'input': input_tensor})
    
    # FPS 계산 및 출력
    fps = 1.0 / (time.time() - start)
    cv2.putText(frame, f"ORIN FP16 FPS: {fps:.1f}", (20, 50), 
                cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
    
    cv2.imshow("Orin Nano Optimized", frame)
    if cv2.waitKey(1) == ord('q'): break

cap.release()
cv2.destroyAllWindows()

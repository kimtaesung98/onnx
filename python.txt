import torch
import torch.nn as nn
from modules.xfeat import XFeat

# 1. 모델 로드 (안정성을 위해 CPU로 로드)
xfeat_api = XFeat()
model = xfeat_api.net.eval().cpu() # .cuda() 대신 .cpu() 사용

# 2. 더미 입력 설정 (CPU 전용)
dummy_input = torch.randn(1, 1, 480, 640).cpu()

# 3. ONNX Export
print("Exporting XFeat to ONNX on CPU (To avoid Error 12/Device Mismatch)...")
# XFeat Backbone은 보통 3개의 값을 반환하므로 output_names를 맞춰줍니다.
torch.onnx.export(model, 
                  dummy_input, 
                  "xfeat.onnx", 
                  input_names=['input'], 
                  output_names=['feats', 'keypoints', 'heatmap'], # 출력 개수 확인
                  opset_version=12,
                  do_constant_folding=True)

print("SUCCESS: xfeat.onnx created on CPU!")

# 4. FP16 변환 (필요 시)
try:
    from onnxconverter_common import float16
    import onnx
    print("Converting to FP16 for Jetson Orin Nano...")
    model_onnx = onnx.load("xfeat.onnx")
    model_fp16 = float16.convert_float_to_float16(model_onnx)
    onnx.save(model_fp16, "xfeat_fp16.onnx")
    print("SUCCESS: xfeat_fp16.onnx created!")
except ImportError:
    print("Note: Install 'onnxconverter-common' to enable half-precision.")

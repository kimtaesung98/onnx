import torch
import numpy as np
from modules.xfeat import XFeat

# 1. Initialize Model
model = XFeat().eval().cuda()

# 2. Create Dummy Input (Standard 640x480 for Jetson)
dummy_input = torch.randn(1, 1, 480, 640).cuda()

# 3. Export to ONNX
print("Exporting xFeat to ONNX (FP32)...")
torch.onnx.export(model, dummy_input, "xfeat.onnx", 
                  input_names=['input'], output_names=['output'],
                  opset_version=12)

# 4. Optional: FP16 Conversion (Using ONNX Runtime tools)
# If you have onnxconverter-common installed:
# pip3 install onnxconverter-common
try:
    from onnxconverter_common import float16
    import onnx
    print("Converting to FP16 for Jetson Nano speed boost...")
    model_onnx = onnx.load("xfeat.onnx")
    model_fp16 = float16.convert_float_to_float16(model_onnx)
    onnx.save(model_fp16, "xfeat_fp16.onnx")
    print("SUCCESS: xfeat_fp16.onnx created!")
except ImportError:
    print("Note: Install 'onnxconverter-common' to enable half-precision.")
